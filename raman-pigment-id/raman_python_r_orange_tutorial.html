# 1. 載入 + 3. 找峰 + 4. 比對(核心)
import numpy as np

def modpoly(x, y, order=5, it=24):        # 2. ModPoly 基線
    yw = y.copy()
    for _ in range(it):
        c = np.polyfit(x, yw, order)
        yw = np.minimum(yw, np.polyval(c, x))
    return np.polyval(np.polyfit(x, yw, order), x)

def pick_peaks(x, y, min_h=0.05, min_dist=9):  # 3. 找峰
    idx = np.where((y[1:-1]>=y[:-2]) & (y[1:-1]>y[2:])
                   & (y[1:-1]>=min_h))[0]+1
    idx = idx[np.argsort(-y[idx])]; keep=[]
    for i in idx:
        if all(abs(x[i]-x[j])>=min_dist for j in keep): keep.append(i)
    return sorted(keep)

# 1. 載入
ref  <- read.csv("raman_reference_spectra.csv", check.names=FALSE)
wn   <- ref[["wavenumber_cm-1"]]; cupc <- ref$CuPc; fcf <- ref$FCF

modpoly <- function(x, y, order=5, it=24) {   # 2. ModPoly 基線
  yw <- y; xc <- scale(x)
  for (i in seq_len(it)) yw <- pmin(yw, predict(lm(yw ~ poly(xc, order))))
  predict(lm(yw ~ poly(xc, order)))
}

pick_peaks <- function(x, y, min_h=0.05, min_dist=9) {  # 3. 找峰
  n <- length(y)
  is_max <- c(FALSE, y[2:(n-1)]>=y[1:(n-2)] & y[2:(n-1)]>y[3:n], FALSE)
  cand <- which(is_max & y>=min_h); cand <- cand[order(-y[cand])]
  keep <- integer(0)
  for (i in cand) if (all(abs(x[i]-x[keep])>=min_dist)) keep <- c(keep,i)
  sort(keep)
}

# 4. 比對(相關係數)
scores <- sapply(list(CuPc=cupc, FCF=fcf), function(r) cor(sample, r))
verdict <- names(which.max(scores))    # -> "CuPc"

先安裝 Spectroscopy 外掛(Options → Add-ons → Spectroscopy),然後拉出這條工作流程:

  1. File — 載入 raman_spectral_library.csv。開 domain editor 把 class 設為 target、其餘波數欄設為 feature(連續)。
  2. File → Preprocess Spectra — 依序加 Baseline (Polynomial) 去背景、Normalize (Vector / Area) 正規化;相當於 Python 的步驟 2。
  3. 接 Spectra viewer — 疊圖看 CuPc 與 FCF 兩群的差異(對應圖 01 的並排比較)。
  4. 接 PCA — 看前兩個主成分能不能把兩類分開(步驟 5 的降維)。
  5. PCA → Test and Score + kNN(或 Logistic Regression),交叉驗證選 5-fold — 得到正確率;對應 Python 的 cross_val_score。
  6. 另接 Predictions + Data Table — 對「未知」樣品輸出預測類別與機率,對應步驟 4 的判定。

流程圖:File → Preprocess Spectra → PCA → Test and Score(kNN),並從 Preprocess 分岔到 Spectra viewer 做視覺檢查。

實際輸出(範例未知樣品):相關係數 CuPc = 0.984,FCF = 0.014 → 判定 CuPc;PCA(5)+kNN 5-fold 正確率 = 1.00。
